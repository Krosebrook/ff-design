1. Supabase schema
create table agent_runs (
  id uuid primary key default gen_random_uuid(),
  org_id uuid not null,
  user_id uuid not null,
  status text check (status in ('queued','running','waiting_approval','failed','completed','cancelled')),
  input jsonb not null,
  output jsonb,
  model_policy jsonb default '{}',
  autonomy_level text check (autonomy_level in ('manual','semi_autonomous','autonomous')),
  cost_cents integer default 0,
  cost_ceiling_cents integer default 50000,
  token_usage jsonb default '{}',
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

create table tool_calls (
  id uuid primary key default gen_random_uuid(),
  run_id uuid references agent_runs(id) on delete cascade,
  tool_name text not null,
  risk_level text check (risk_level in ('low','medium','high','critical')),
  input jsonb not null,
  output jsonb,
  status text check (status in ('pending','approved','denied','running','failed','completed')),
  idempotency_key text not null unique,
  requires_approval boolean default false,
  cost_estimate_cents integer,
  created_at timestamptz default now()
);

create table approvals (
  id uuid primary key default gen_random_uuid(),
  run_id uuid references agent_runs(id) on delete cascade,
  tool_call_id uuid references tool_calls(id) on delete cascade,
  status text check (status in ('pending','approved','denied')),
  requested_by uuid not null,
  reviewed_by uuid,
  reason text,
  created_at timestamptz default now(),
  reviewed_at timestamptz
);

create table audit_logs (
  id uuid primary key default gen_random_uuid(),
  org_id uuid not null,
  actor_type text check (actor_type in ('human','agent','system')),
  action text not null,
  target_type text not null,
  target_id text,
  metadata jsonb default '{}',
  created_at timestamptz default now()
);

create index idx_runs_org on agent_runs(org_id);
create index idx_runs_user on agent_runs(user_id);
create index idx_tool_calls_run on tool_calls(run_id);
create index idx_approvals_run on approvals(run_id);
2. LangGraph orchestrator (Python)
from langgraph.graph import StateGraph
from typing import TypedDict, Any
import anthropic
import json

class AgentState(TypedDict):
    run_id: str
    org_id: str
    user_id: str
    input_task: str
    cost_cents: int
    cost_ceiling: int
    tool_calls: list
    output: Any

async def intake_node(state: AgentState):
    state['cost_cents'] += 5
    return state

async def plan_node(state: AgentState):
    client = anthropic.Anthropic()
    msg = client.messages.create(
        model="claude-opus-4-20250514",
        max_tokens=500,
        messages=[{"role": "user", "content": state['input_task']}]
    )
    state['cost_cents'] += int(msg.usage.input_tokens * 0.05 / 100)
    return state

async def execute_node(state: AgentState):
    if state['cost_cents'] >= state['cost_ceiling']:
        raise Exception(f"Cost ceiling ${state['cost_ceiling']/100} exceeded")
    return state

async def finalize_node(state: AgentState):
    return state

workflow = StateGraph(AgentState)
workflow.add_node("intake", intake_node)
workflow.add_node("plan", plan_node)
workflow.add_node("execute", execute_node)
workflow.add_node("finalize", finalize_node)

workflow.add_edge("intake", "plan")
workflow.add_edge("plan", "execute")
workflow.add_edge("execute", "finalize")

app = workflow.compile()
3. Next.js API handler
// app/api/runs/route.ts
import { createClient } from '@supabase/supabase-js'
import { NextRequest, NextResponse } from 'next/server'

const supabase = createClient(
  process.env.SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_KEY!
)

export async function POST(req: NextRequest) {
  try {
    const { org_id, user_id, task, autonomy_level = 'semi_autonomous', cost_ceiling_cents = 50000 } = await req.json()
    
    const { data: run, error } = await supabase
      .from('agent_runs')
      .insert({
        org_id,
        user_id,
        input: { task },
        status: 'queued',
        autonomy_level,
        cost_ceiling_cents,
        model_policy: {}
      })
      .select('id')
      .single()

    if (error) throw error

    return NextResponse.json({ run_id: run.id }, { status: 201 })
  } catch (err) {
    return NextResponse.json(
      { error: String(err) },
      { status: 400 }
    )
  }
}

export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url)
  const org_id = searchParams.get('org_id')

  const { data: runs } = await supabase
    .from('agent_runs')
    .select('*')
    .eq('org_id', org_id)
    .order('created_at', { ascending: false })
    .limit(20)

  return NextResponse.json(runs)
}

RLS policies (Supabase SQL)
-- Enable RLS
alter table agent_runs enable row level security;
alter table tool_calls enable row level security;
alter table approvals enable row level security;
alter table audit_logs enable row level security;

-- Users can only see their org's runs
create policy "Users see own org runs" on agent_runs
  for select using (auth.uid() = user_id);

create policy "Users insert own runs" on agent_runs
  for insert with check (
    auth.uid() = user_id and 
    org_id in (
      select org_id from user_orgs where user_id = auth.uid()
    )
  );

-- Tool calls follow run visibility
create policy "Tool calls inherit run visibility" on tool_calls
  for select using (
    run_id in (
      select id from agent_runs where user_id = auth.uid()
    )
  );

-- Approvals only visible to reviewers
create policy "Approvals visible to reviewers" on approvals
  for select using (
    reviewed_by = auth.uid() or 
    requested_by = auth.uid()
  );

-- Audit logs only visible to org admins
create policy "Audit logs org-scoped" on audit_logs
  for select using (
    org_id in (
      select org_id from users where id = auth.uid()
    )
  );
